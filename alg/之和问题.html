# 两数之和

这个是找索引，所以不能用双指针（双指针要求数组有序，但是一旦排序就不能得到原来的索引了）

class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> map = new HashMap<>();
        for (int i = 0; i < nums.length; i++) {
            if (map.containsKey(target - nums[i])) {
                return new int[]{map.get(target - nums[i]), i};
            }
            map.put(nums[i], i);
        }

        return null;
    }
}

# 三数之和

class Solution {
    public List<List<Integer>> threeSum(int[] nums) {
        return threeSum(nums, 0);
    }


    public List<List<Integer>> threeSum(int[] nums, int begin, int target) {
        var l = nums.length;
        var res = new ArrayList<List<Integer>>();
        for (int i = begin; i < l - 2; i++) {
            if (i == begin || nums[i] != nums[i - 1]) {
                final int v = nums[i];
                var twoSumList = twoSum(nums, i, target - v);
                twoSumList.forEach(list -> {
                    list.add(v);
                    res.add(list);
                });
            }
        }
        return res;
    }

    private List<List<Integer>> twoSum(int[] nums, int begin, int target) {
        int left, right, start, end;
        start = left = begin + 1;
        end = right = nums.length - 1;
        var res = new ArrayList<List<Integer>>();
        while (left < right) {
            //跳过相同的
            if (left != start && nums[left] == nums[left - 1]) {
                left++;
                continue;
            }
            if (right != end && nums[right] == nums[right + 1]) {
                right--;
                continue;
            }
            var sum = nums[left] + nums[right];
            if (sum > target) {
                right--;
            } else if (sum == target) {
                var list = new ArrayList<Integer>();
                list.add(nums[left]);
                list.add(nums[right]);
                res.add(list);
                left++;
                right--;
            } else {
                left++;
            }

        }
        return res;
    }
}

# 四数之和

给三数之和套一层

class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        var res = new ArrayList<List<Integer>>();
        Arrays.sort(nums);

        for (int i = 0; i < nums.length - 3; i++) {
            final int v = nums[i];
            if (i == 0 || v != nums[i - 1]) {
                var threeSumRes = threeSum(nums, i + 1, target - v);
                threeSumRes.forEach(list -> {
                    list.add(v);
                    res.add(list);
                });
            }
        }
        return res;
    }

# n 数之和

从三数之和、四数之和可以找到规律，在 n 大于 2 时循环递归，等于 2 时双指针寻找

class Solution {

    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        return nSum(nums, 4, 0, target);
    }

    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        return nSum(nums, 3, 0, 0);
    }

    public List<List<Integer>> nSum(int[] nums, int n, int begin, int target) {
        var l = nums.length;

        var res = new ArrayList<List<Integer>>();

        if (n == 2) {
            int left, right, start, end;
            start = left = begin;
            end = right = l - 1;
            while (left < right) {
                //跳过相同的
                if (left != start && nums[left] == nums[left - 1]) {
                    left++;
                    continue;
                }
                if (right != end && nums[right] == nums[right + 1]) {
                    right--;
                    continue;
                }
                var sum = nums[left] + nums[right];
                if (sum > target) {
                    right--;
                } else if (sum == target) {
                    var list = new ArrayList<Integer>();
                    list.add(nums[left]);
                    list.add(nums[right]);
                    res.add(list);
                    left++;
                    right--;
                } else {
                    left++;
                }
            }
        } else {
            for (int i = begin; i < l - n + 1; i++) {
                if (i == begin || nums[i] != nums[i - 1]) {
                    final int v = nums[i];
                    nSum(nums, n - 1, i + 1, target - v).forEach(list -> {
                        list.add(v);
                        res.add(list);
                    });
                }
            }
        }
        return res;
    }
}
