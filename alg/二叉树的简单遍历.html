class TraverseBinaryTree {

    void preOrder(TreeNode root) {
        if (root == null) return;
        preOrder(root.left);
        System.out.println(root);
        preOrder(root.right);
    }

    void inOrder(TreeNode root) {
        if (root == null) return;
        System.out.println(root);
        inOrder(root.left);
        inOrder(root.right);
    }

    void postOrder(TreeNode root) {
        if (root == null) return;
        postOrder(root.left);
        postOrder(root.right);
        System.out.println(root);
    }

    // 层序遍历
    void index(TreeNode root) {
        LinkedList<TreeNode> q = new LinkedList<>();
        q.add(root);
        while (!q.isEmpty()) {
            var node = q.remove();
            System.out.println(node.val);
            if (node.left != null)
                q.add(node.left);
            if (node.right != null)
                q.add(node.right);
        }
    }

    // 二叉树的右侧视图
    List<Integer> rightSideView(TreeNode root) {
        var map = new TreeMap<Integer, Integer>();
        var res = new ArrayList<Integer>();
        travel(root, 1, map);
        return new ArrayList<>(map.values());
    }

    void travel(TreeNode node, int index, Map<Integer, Integer> cache) {
        if (node == null) return;
        cache.putIfAbsent(index, node.val);
        travel(node.right, index + 1, cache);
        travel(node.left, index + 1, cache);
    }

    List<Integer> rightSideViewIndex(TreeNode root) {
        if(root == null) return List.of();

        Map<Integer, Integer> rightmostValueAtDepth = new TreeMap<>();

        // 采用双栈
        Queue<TreeNode> nodeQueue = new ArrayDeque<>();
        Queue<Integer> depthQueue = new ArrayDeque<>();

        // 初始化
        nodeQueue.add(root);
        depthQueue.add(0);

        while (!nodeQueue.isEmpty()) {
            TreeNode node = nodeQueue.remove();
            int depth = depthQueue.remove();

            // 由于每一层最后一个访问到的节点才是我们要的答案，因此不断更新对应深度的信息即可
            rightmostValueAtDepth.put(depth, node.val);

            if (node.left != null) {
                nodeQueue.add(node.left);
                depthQueue.add(depth + 1);
            }
            if (node.right != null) {
                nodeQueue.add(node.right);
                depthQueue.add(depth + 1);
            }
        }

        return new ArrayList<>(rightmostValueAtDepth.values());
    }
    
    // 翻转二叉树
    TreeNode invertTree(TreeNode root) {
        if (root == null) return root;
        var temp = root.right;
        root.right = invertTree(root.left);
        root.left = invertTree(temp);
        return root;
    }
}
