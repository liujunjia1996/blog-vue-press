摩尔投票时是一种同归于尽的思想，换句话说就是，利用目标元素是多数的特点，让非目标元素和目标元素进行 ”一换一“ 的一种思想。
有两个题用了摩尔投票：

# 多数元素open in new window

class Solution {
    public int majorityElement(int[] nums) {
        // 使用摩尔投票法，只需遍历一次即可
        var res = nums[0];
        var times = 1;
        for (int i = 1; i < nums.length; i++) {
            if (times == 0) {
                res = nums[i];
                times = 1;
                continue;
            }
            times = nums[i] == res ? times + 1 : times - 1;
        }
        return res;
    }
}

# 求众数open in new window

class Solution {
    public List<Integer> majorityElement(int[] nums) {
        // analyze：must be 2 or 1 num
        var l = nums.length;
        int num1 = Integer.MAX_VALUE, num2 = Integer.MAX_VALUE;
        var count1 = 0;
        var count2 = 0;
        for (int num : nums) {
            if (count1 == 0 && num != num2) {
                count1 = 1;
                num1 = num;
            } else if (count2 == 0 && num != num1) {
                count2 = 1;
                num2 = num;
            } else if (num == num1) {
                count1++;
            } else if (num == num2) {
                count2++;
            } else {
                count1--;
                count2--;
            }
        }
        var newCount1 = 0;
        var newCount2 = 0;
        for (int num : nums) {
            if (num == num1) newCount1++;
            if (num == num2) newCount2++;
        }
        var res = new ArrayList<Integer>();
        if (newCount1 * 3 > l) {
            res.add(num1);
        }
        if (newCount2 * 3 > l) {
            res.add(num2);
        }
        return res;
    }
}
