递增子序列open in new window

这个题挺经典的，dp [i] 的定义为以 nums [i] 结尾的最长递增子序列的长度；
这样一定义状态就能比较简单的转移了。

/**
 * @param {number []} nums
 * @return {number}
 */
var lengthOfLIS = function (nums) {
    var n = nums.length;
    if (n === 0 || n === 1) {
        return n;
    }
    // dp [i] 定义为以 nums [i] 结尾的子序列长度
    var dp = new Array(n).fill(1);
    var max = 1;
    for (var i = 1; i < n; i++) {
        for (var j = 0; j < i; j++) {
            if (nums[i] > nums[j]) {
                dp[i] = Math.max(dp[i], dp[j] + 1);
            }
        }
        max = Math.max(max, dp[i]);
    }
    return max;
};
