# awk

主要用于切割文本

#!/bin/sh

useRateStr=`df -h | grep /dev/vda1 | awk  '{print $5}'`
useRate=${useRateStr: 0: 2}

cd `find /opt -maxdepth 3 -name nacos | awk 'NR==1'`/bin/logs

while [ $useRate -gt 40 ]; do
    ls -l ./ | awk 'NR==2' | awk  '{print $9}' | xargs rm -f
    useRateStr=`df -h | grep /dev/vda1 | awk  '{print $5}'`
    useRate=${useRateStr: 0: 2}
done
